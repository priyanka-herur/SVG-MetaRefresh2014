CSS 
.bird .right_wing
{
 -webkit-animation-name: flapright;
 -webkit-animation-duration: 0.05s; 
 -webkit-animation-iteration-count: infinite;
 -webkit-animation-timing-function: ease-in-out;
 -webkit-animation-direction: alternate;
 -webkit-transform-origin: 50%;
}   
@-webkit-keyframes flapright { 
 0% { -webkit-transform:translate(3px,1px)}
 100% { -webkit-transform: translate(0px,5px)}
}

LINE DRAWING

Draw 
<path stroke-dasharray="3000,3000"  
        stroke-dashoffset="3309"  d="M ..." 
        stroke="#333" stroke-width="1" fill="none">
 <animate attributeName="stroke-dashoffset" 
        from="3309" to="7s" dur="8s" 
        begin="startLineDraw.click" 
        fill="freeze"/>
</path>

ANIMATED TEXT ON PATH

Draw 
//Define your path
<path id="curve" d="M ... " stroke="#000000"/>

//Get your text on the path
<text xmlns="http://www.w3.org/2000/svg" >
<textPath xlink:href="#curve" 
        xmlns:xlink="http://www.w3.org/1999/xlink" 
        startOffset="10">
<animate attributeName="startOffset" dur="20s" 
     from="0" to="1500" repeatCount="indefinite"/>
        Featured location: Dominica
</textPath>
</text>
